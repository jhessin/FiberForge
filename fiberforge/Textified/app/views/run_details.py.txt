from textual.widgets import Static

from py_tools.design import FiberRun, LengthService


class RunDetails(Static):

    def update_run(self, run: FiberRun) -> None:
        total = LengthService.total(run)
        slack = LengthService.slack(run)
        riser = LengthService.riser(run)

        cans = [span.start_can.value for span in run.spans]
        cans.append(run.spans[-1].end_can.value)

        spans = "\n".join(f"{span.id.value} ({span.size}ct)" for span in run.spans)

        self.update(
            f"Job: {run.job_id.value}\n\n"
            f"[b]Total Length:[/b] {total}\n"
            f"[b]Slack:[/b] {slack}\n"
            f"[b]Riser:[/b] {riser}\n\n"
            f"[b]Cans:[/b]\n"
            + "\n".join(f"  - {c}" for c in cans)
            + "\n\n[b]Spans:[/b]\n"
            + spans
        )
